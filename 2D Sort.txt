void Sort(int niz[][20]) {

	int privremena = 0;
	bool izmjena = false;
	do {
		for (int i = 0; i < 20; i++)
		{
			if (niz[i][kolona] > niz[i + 1][kolona])
			{
				ZamjeniRed(niz, i, i + 1);

			}

}
void ZamjeniRed(int niz[][20],int a,int b) {

		temp = niz[a][i];
		niz[a][i] = niz[b][i];
		niz[b][i] = temp;
	}
}
